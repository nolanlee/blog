Promise.allPolyfill = function(arr) {
  const result = []
  let count = 0
  return new Promise((resolve, reject) => {
    arr.forEach((item, index) => {
      Promise.resolve(item).then(res => {
        result[index] = res
        count++
        if(count === arr.length) resolve(result)
      }).catch(reject)
    })
  })
}
